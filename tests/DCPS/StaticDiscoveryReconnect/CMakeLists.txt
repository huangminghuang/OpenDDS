project(StaticDiscoveryReconnectTest CXX)
cmake_minimum_required(VERSION 3.3)

if (NOT OpenDDS_INCLUDE_DIR)
  find_package(OpenDDS REQUIRED CONFIG)
endif()


ace_add_exe(StaticDiscoveryReconnectTest
  OUTPUT_NAME StaticDiscoveryTest
  LINK_LIBRARIES OpenDDS_Transports_For_Test
                 OpenDDS_Default_Discovery
)
ace_target_cxx_sources(StaticDiscoveryReconnectTest
  SOURCE_FILES StaticDiscoveryTest.cpp
)

dds_idl_sources(
  TARGETS StaticDiscoveryReconnectTest
  TAO_IDL_FLAGS -SS
  IDL_FILES TestMsg.idl
)

include(${OpenDDS_INCLUDE_DIR}/cmake/dds_add_test.cmake)
dds_configure_test_files()


dds_add_test("tests/DCPS/StaticDiscoveryReconnect/run_test.pl writer_first"
  COMMAND run_test.pl writer_first
  LABELS MCAST
)

dds_add_test("tests/DCPS/StaticDiscoveryReconnect/run_test.pl reader_first"
  COMMAND run_test.pl reader_first
  LABELS MCAST
)
